lineari mainigais signals

%k=(yA-yB)/(tA-tB)

t_saw= 4:0.01:5.5;
k=(1.5+1.5)/(4-5.5);
delay=4.75;
y_saw=k*(t_saw-delay);

sinusoida

t_sin=5.5:0.01:7.5;
a0=0; a=0.5; T=(3.5-1)/2;f=2.81/T;
delay=1.5;
y_sin=a0+a*sin(2*pi*f*(t_sin-delay));
